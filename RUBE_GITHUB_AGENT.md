```python
import requests
import json
from datetime import datetime, timedelta

# GitHub API

GITHUB_TOKEN = "YOUR_GITHUB_TOKEN"

# FrogFace API
FROGFACE_API = "https://frogface-rpg.vercel.app/api/tasks"

def get_new_issues():
    """–ü–æ–ª—É—á–∏—Ç—å –Ω–æ–≤—ã–µ issues –∑–∞ –ø–æ—Å–ª–µ–¥–Ω–∏–µ 24 —á–∞—Å–∞"""
    url = f"https://api.github.com/repos/{GITHUB_REPO}/issues"
    headers = {"Authorization": f"token {GITHUB_TOKEN}"}
    params = {
        "state": "open",
        "sort": "created",
        "direction": "desc",
        "since": (datetime.now() - timedelta(days=1)).isoformat()
    }
    
    response = requests.get(url, headers=headers, params=params)
    return response.json()

def determine_priority(labels):
    """–û–ø—Ä–µ–¥–µ–ª–∏—Ç—å –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç –ø–æ –ª–µ–π–±–ª–∞–º"""
    priority_map = {
        'bug': 'high',
        'critical': 'high',
        'enhancement': 'medium',
        'documentation': 'low',
        'feature': 'high'
    }
    
    for label in labels:
        label_name = label.get('name', '').lower()
        if label_name in priority_map:
            return priority_map[label_name]
    
    return 'medium'  # –ü–æ —É–º–æ–ª—á–∞–Ω–∏—é

def determine_project(title, body):
    """–û–ø—Ä–µ–¥–µ–ª–∏—Ç—å –ø—Ä–æ–µ–∫—Ç –ø–æ —Å–æ–¥–µ—Ä–∂–∏–º–æ–º—É"""
    text = (title + ' ' + body).lower()
    
    if any(word in text for word in ['edison', '–±–∞—Ä', '—Ä–µ—Å—Ç–æ—Ä–∞–Ω']):
        return 'edison'
    if any(word in text for word in ['receptor', 'saas', 'b2b']):
        return 'receptor'
    if any(word in text for word in ['frogface', 'rpg', '–≥–µ–π–º–∏—Ñ–∏–∫–∞—Ü–∏—è']):
        return 'frogface'
    
    return 'personal'

def create_quest_in_frogface(issue):
    """–°–æ–∑–¥–∞—Ç—å –∫–≤–µ—Å—Ç –≤ FrogFace –∏–∑ GitHub issue"""
    title = issue['title']
    body = issue.get('body', '')
    labels = issue.get('labels', [])
    
    priority = determine_priority(labels)
    project = determine_project(title, body)
    
    payload = {
        "title": title,
        "priority": priority,
        "projectId": project
    }
    
    response = requests.post(FROGFACE_API, json=payload)
    return response.json()

def main():
    """–û—Å–Ω–æ–≤–Ω–∞—è –ª–æ–≥–∏–∫–∞ –∞–≥–µ–Ω—Ç–∞"""

        # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º PR (—É –Ω–∏—Ö –µ—Å—Ç—å pull_request –ø–æ–ª–µ)
        if 'pull_request' in issue:
            continue
        
        # –°–æ–∑–¥–∞–µ–º –∫–≤–µ—Å—Ç –≤ FrogFace

        print(f"‚úÖ –ö–≤–µ—Å—Ç —Å–æ–∑–¥–∞–Ω –∏–∑ issue #{issue['number']}: {issue['title']}")
        print(f"   ‚Üí FrogFace Task ID: {result.get('task', {}).get('id')}")

if __name__ == "__main__":
    main()
```
